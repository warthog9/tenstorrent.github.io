102template<EltwiseBinaryType tBcastOp, BroadcastType tBcastDim>
103void init_bcast(uint32_t icb0, uint32_t icb1, uint32_t ocb = 16)
104{
105 if constexpr (tBcastOp == ELWMUL)
106 MATH(( llk_math_eltwise_binary_init<tBcastOp, tBcastDim, MATH_FIDELITY>() ));
107 else
108 MATH(( llk_math_eltwise_binary_init<tBcastOp, tBcastDim>() ));
109
110 UNPACK(( llk_setup_operands() ));
111 UNPACK(( llk_unpack_AB_hw_configure_disaggregated<DST_ACCUM_MODE>(icb0, icb1) ));
112 UNPACK(( llk_unpack_AB_init<tBcastDim>(icb0, icb1) ));
113 // TODO(AP): running this specific init after common AB init causes a hang
114
115 // clone of general init for AB TODO(AP): commonize
116 //UNPACK(( llk_setup_operands() ));
117 //UNPACK(( llk_unpack_AB_init<BroadcastType::NONE>() ));
118 //UNPACK(( llk_unpack_AB_hw_configure_disaggregated<BroadcastType::NONE>(icb0, icb1) ));
119
120 PACK(( llk_pack_hw_configure_disaggregated<false, DST_ACCUM_MODE>(ocb) ));
121 PACK(( llk_pack_init(ocb) ));
122 PACK(( llk_setup_outputs() ));
123 PACK(( llk_pack_dest_init<false, DST_ACCUM_MODE>() ));
124
125 MATH(( llk_math_pack_sync_init<DST_ACCUM_MODE>() ));
126}
