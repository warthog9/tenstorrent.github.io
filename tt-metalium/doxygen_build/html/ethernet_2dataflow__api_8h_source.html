112FORCE_INLINE
113void eth_send_bytes(
114 uint32_t src_addr,
115 uint32_t dst_addr,
116 uint32_t num_bytes,
117 uint32_t num_bytes_per_send = 16,
118 uint32_t num_bytes_per_send_word_size = 1) {
119 uint32_t num_bytes_sent = 0;

123 num_bytes_sent += num_bytes_per_send;
124 }
125 erisc_info->channels[0].bytes_sent += num_bytes;
126}

397FORCE_INLINE
398void eth_wait_for_bytes_on_channel(uint32_t num_bytes, uint8_t channel) {
399 // assert(channel < 4);
400 uint32_t count = 0;
401 uint32_t poll_count = 1000000;
402 uint32_t num_bytes_sent = erisc_info->channels[channel].bytes_sent;
403 while (num_bytes_sent == 0) {
404 uint32_t received_this_iter = erisc_info->channels[channel].bytes_sent;
405 if (received_this_iter != num_bytes_sent) {
406 // We are currently in the process of receiving data on this channel, so we just just wait a
407 // bit longer instead of initiating a context switch
408 num_bytes_sent = received_this_iter;
409 } else {
410 count++;
411 if (count > poll_count) {
412 count = 0;
413 run_routing();
414 }
415 }
416 }
417}
