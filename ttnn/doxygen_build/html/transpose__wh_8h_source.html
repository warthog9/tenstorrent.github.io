27ALWI void transpose_wh_init(uint32_t icb, uint32_t ocb = 16)
28{
29 MATH(( llk_math_eltwise_unary_datacopy_init<A2D, BroadcastType::NONE, DST_ACCUM_MODE>(true, true, icb) ));
30 MATH(( llk_math_pack_sync_init<DST_ACCUM_MODE>() ));
31
32 PACK(( llk_pack_hw_configure_disaggregated<false, DST_ACCUM_MODE>(ocb) ));
33 PACK(( llk_pack_init(ocb) ));
34 PACK(( llk_setup_outputs() ));
35 PACK(( llk_pack_dest_init<false, DST_ACCUM_MODE>() ));
36

38 UNPACK(( llk_unpack_A_hw_configure_disaggregated<DST_ACCUM_MODE>(0, true) ));
39 UNPACK(( llk_unpack_A_init<BroadcastType::NONE, true, EltwiseBinaryReuseDestType::NONE>(true, true) ));
40}
